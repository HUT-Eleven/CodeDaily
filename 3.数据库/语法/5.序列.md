```
--创建默认的序列(点击view SQL可以查看SQL语句)
create sequence  序列名;
```

```
--查看当前序列
select 序列名.currval from dual;
```

```
--查询下一个序列
select  序列名.nextval from dual;
```

```
--创建指定参数的序列（不用逗号隔开，就是一条语句，后面是设置属性，顺序不唯一）
create  sequence seq_dept 
start with  10  
maxvalue  100000 
cycle   --或者nocycle
increment by 10 --增量值 
cache  50;--缓冲量
```

```
--删除序列
drop  sequence seq_dept;
```

```
--insert 插入:（把序列插入某一列，该列要是数字类型）
         --语法：
                insert  into  表名 values(列值1,列值2....);
         
--往部门表中插入数据
    insert  into  dept values(seq_dept.nextval,'教学部');              
    insert  into  dept values(seq_dept.nextval,'市场部');
```
