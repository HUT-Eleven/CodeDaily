CREATE OR REPLACE PROCEDURE addnum(v_a NUMBER,
                                   v_b IN NUMBER,
                                   v_c OUT NUMBER,
                                   v_d IN OUT NUMBER)
IS

BEGIN
    IF(v_a>v_b) THEN
        v_c:=v_a;
    ELSE
        v_c:=v_b;
    END IF;
    v_d:=v_d+1;
END;

--调用该过程

END;
