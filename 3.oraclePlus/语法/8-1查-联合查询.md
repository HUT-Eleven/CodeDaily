/*
        并集：
               union  all ：把两个查询结果连接在一起，会有重复的记录
               

```
           union  :把两个查询结果连接在一起，会把结果去重复
```

*/
--查询出所有的员工信息
select * from  emp
union  all
select * from emp_bak;

select * from  emp
union

/*     交集：
       intersect ：把两个查询结果相交的显示出来
*/

intersect  
select * from emp_bak;

/*
       补集：minus:减去查询结果B的数据，并且减去A跟B相交的结果，显示剩下的结果
*/
