```
	--外连接（左外和右外都有两种方式，全只有一种。left/right/full join ... on..）
--查询出所有的部门，以及部门下面的员工信息
select * from dept d right join emp e on d.deptno = e.deptno;

select * from dept d ,emp e where d.deptno(+)=e.deptno;
(+号在左为“右外连接”，反之。没有+号的表示可以取到本身可以取到的全部的值，而有+号的就表示需要去补null去青睐)

--全连接
select  * from  emp e full join  dept d  on e.deptno= d.deptno;
```

```
--自连接：把自己本身看作成多张表，自己跟自己进行连接
--查询员工编号，姓名，上级编号，上级姓名

select e.empno 员工编号,
       e.ename 员工姓名,
       m.empno 上面编号,
       m.ename 上级姓名
  from emp e, emp m
 where e.mgr = m.empno(+);
```

```
--交叉连接：
select * from  emp e cross join dept d;
```
