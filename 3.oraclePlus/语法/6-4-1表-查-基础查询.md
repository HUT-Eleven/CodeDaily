--1.基础查询：

```
--查询出所有员工编号，姓名，薪资
(这里的薪资是第一个要加双引号的地方，因为中间有空格。as可有可无)
select empno 员工编号,ename as 员工姓名,sal  "薪  资",e.hiredate from emp e;
```



```
--查询出所有员工编号，姓名，薪资，拼接在一列显示
select '员工编号：'||empno || '姓名：'|| ename || '薪资：'|| sal  员工信息 from emp; 
（||表示拼接，列和表都可以取别名）
```

```
--查询出有哪几种岗位（distinct去重复）
select  distinct job from emp;
```

```
--查询出所有的员工信息，根据薪资进行降序排序（order by）
--desc ：降序   asc：升序   默认升序
select * from  emp  order by  sal desc;
```

--2.查询中的算术运算符    

```
 日期类型跟数值类型相加或相减的是天数
 日期跟日期只能相减，得到是天数。日期不能相加。
```

```
--查询出所有员工的编号，姓名，job，sal,入职日期，转正日期
select empno ,ename,job, sal ,hiredate 入职日期,hiredate +90 转正日期 from emp;
```

```
--查询出所有员工的编号，姓名，job，sal,入职日期，工作年限
select empno,ename,job,sal,hiredate 入职日期,trunc((sysdate -hiredate)/365)  工作年限 from emp;
```

--3.比较运算符



```
--查询出不等于20号部门下面的员工（！=和<>都表示不等）
select * from emp where deptno != 20;
select * from emp where deptno <> 20;
```

```
--查询出’1981-9-1‘后入职的员工信息
select * from  emp  where  hiredate  > to_date('1981-09-01','yyyy-mm-dd') ;
```

```
--查询出在 ’1981-11-01‘  至 ’1987-04-01‘入职员工信息
select *  from emp where hiredate >= to_date('1981-11-01', 'yyyy-mm-dd')   and hiredate <= to_date('1987-4-1', 'yyyy-mm-dd');

select *  from emp where hiredate between to_date('1981-11-01', 'yyyy-mm-dd') and  to_date('1987-4-1', 'yyyy-mm-dd');
```

```
--查询出没有上级的员工信息（is null表示为null，is not null反之这的is不能舍去）
select * from emp where mgr is  null;
```

```
--查询20,30，部门下面的员工
select * from  emp  where deptno =20 or  deptno =30;

select * from  emp where deptno in(20,30);
```



--4.匹配符(模糊查询)

```
--查询出员工姓名首字母为’S'的员工信息
--   _ ：代表一个任意字符    % ：代表任意个任意字符
select * from  emp  where  ename like 'S%';
```

```
--查询员工姓名包含‘O' 的员工信息
select * from  emp where ename like '%O%';
```

```
--查询员工姓名第三个字符为’R‘ 的员工信息
select *from  emp where ename like'__R%';
```
